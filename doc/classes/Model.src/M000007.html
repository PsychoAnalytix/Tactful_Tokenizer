    def get_features(frag, model)
        w1 = (frag.cleaned.split.last or '')
        w2 = (frag.next.andand.first or '')

        frag.features = ["w1_#{w1}""w1_#{w1}", "w2_#{w2}""w2_#{w2}", "both_#{w1}_#{w2}""both_#{w1}_#{w2}"]

        len1 = [10, w1.gsub(/\W/, '').length].min

        if not w2.empty? and w1.gsub(/\./, '').is_alphabetic? 
            frag.features.push "w1length_#{len1}""w1length_#{len1}"
            begin
                frag.features.push "w1abbr_#{Math.log(1 + model.non_abbrs[w1.chop]).to_i}""w1abbr_#{Math.log(1 + model.non_abbrs[w1.chop]).to_i}"
            rescue Exception => e
                frag.features.push :"w1abbr_0"
            end
        end

        if not w2.empty? and w2.gsub(/\./, '').is_alphabetic?
            frag.features.push "w2cap_#{w2[0].is_upper_case?.to_s.capitalize}""w2cap_#{w2[0].is_upper_case?.to_s.capitalize}"
            begin
                frag.features.push "w2lower_#{Math.log(1 + model.lower_words[w2.downcase]).to_i}""w2lower_#{Math.log(1 + model.lower_words[w2.downcase]).to_i}"
            rescue Exception => e
                frag.features.push :"w2lower_0"
            end
        end
    end
