    def classify_single(frag)
        probs = [@p0, @p1]
        feat = ''
        frag.features.each do |feat|
            probs[0] *= (@feats["0,#{feat}""0,#{feat}"] or next)
            probs[1] *= @feats["1,#{feat}""1,#{feat}"]
        end
        frag.pred = normalize(probs)[1]
    end
