        def initialize(text)
            @frags = []
            res = nil
            puts "Hey!"
            puts text.inspect
            text.each_line do |line|
                unless line.strip.empty?
                    line.split(/(.*?[.!?](?:["')\]}]|(?:<.*>))*[\s])/).each do |res|
                        unless res.strip.empty?
                            frag = Frag.new(res)
                            @frags.last.next = frag.cleaned.first unless @frags.empty?
                            @frags.push frag
                        end
                    end
                end
            end
        end
