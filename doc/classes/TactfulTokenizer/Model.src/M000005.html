        def classify(doc)
            frag = nil
            probs = 1
            feat = ''
            doc.frags.each do |frag|
                probs = @p0
                frag.features.each do |feat|
                    probs *= @feats[feat]
                end
                frag.pred = probs / (probs + 1)
            end
        end
