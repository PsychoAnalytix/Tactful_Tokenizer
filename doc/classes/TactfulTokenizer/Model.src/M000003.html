        def initialize(feats="models/features.mar", lower_words="models/lower_words.mar", non_abbrs="models/non_abbrs.mar")
            @feats, @lower_words, @non_abbrs = [feats, lower_words, non_abbrs].map do |file|
                File.open(file) do |f|
                    Marshal.load(f.read)
                end
            end
            @p0 = @feats["<prior>"] ** 4  
        end
